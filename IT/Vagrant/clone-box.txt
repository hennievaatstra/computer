. go to box-to-clone dir
$ cd /home/Vagrant/rocky-91-01

. package the box
$ vagrant package

. copy the package to the images location
$ sudo cp package.box /home/libvirt/images

. create new box dir
$ cd /home/Vagrant
$ mkdir rocky-91-02
$ cd $_

. setup new machine
$ vagrant init

. point to the new image
  config.vm.box = "rocky-91_default.box"
  config.vm.box_url = "file:///home/libvirt/images/rocky-91_default.box"

. start the box
$ vagrant up
